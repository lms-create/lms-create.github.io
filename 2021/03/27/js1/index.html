数组扁平化

const animals = ["🐷", ["🐶", "🐂"], ["🐎", ["🐑", ["🐲"]], "🐛"]];

// 不传参数时，默认“拉平”一层
animals.flat();
// ["🐷", "🐶", "🐂", "🐎", ["🐑", ["🐲"]], "🐛"]

// 传入一个整数参数，整数即“拉平”的层数
animals.flat(2);
// ["🐷", "🐶", "🐂", "🐎", "🐑", ["🐲"], "🐛"]

// Infinity 关键字作为参数时，无论多少层嵌套，都会转为一维数组
animals.flat(Infinity);
// ["🐷", "🐶", "🐂", "🐎", "🐑", "🐲", "🐛"]

// 传入 <=0 的整数将返回原数组，不“拉平”
animals.flat(0);
animals.flat(-10);
// ["🐷", ["🐶", "🐂"], ["🐎", ["🐑", ["🐲"]], "🐛"]];

// 如果原数组有空位，flat()方法会跳过空位。
["🐷", "🐶", "🐂", "🐎",,].flat();
// ["🐷", "🐶", "🐂", "🐎"]

手动实现flat()函数

思路：首先需要遍历数组中的元素，然后再判断元素是否为数组，再将数组展开。

Concat()就是将数组合并。

const arr = [1, 2, 3, 4, [1, 2, 3, [1, 2, 3, [1, 2, 3]]], 5];
// concat + 递归
function flat(arr) {
  let arrResult = [];
  arr.forEach(item => {
    if (Array.isArray(item)) {
      arrResult = arrResult.concat(arguments.callee(item));   // 递归
      // 或者用扩展运算符
      // arrResult.push(...arguments.callee(item));
    } else {
      arrResult.push(item);
    }
  });
  return arrResult;
}
let lms = flat(arr);
console.log(lms);
// [1, 2, 3, 4, 1, 2, 3, 1, 2, 3, 1, 2, 3, 5];

reduce()实现flat()

reduce的作用是将数组中的元素执行自己定义的回调函数。

例如：

var total = [1,2,3,4].reduce((a,b)=>{
  return a + b;
})
//total = 10

var total = [[1,2],[3,4],[5,6]].reduce((a,b)=>{
  return a.concat(b);
})
//total[1,2,3,4,5,6];

所以我们可以写出以下代码：

const arr = [1, 2, 3, 4, [1, 2, 3, [1, 2, 3, [1, 2, 3]]], 5]

// 首先使用 reduce 展开一层
arr.reduce((pre, cur) => pre.concat(cur), []);
// [1, 2, 3, 4, 1, 2, 3, [1, 2, 3, [1, 2, 3]], 5];

// 用 reduce 展开一层 + 递归
const flat = arr => {
  return arr.reduce((pre, cur) => {
    return pre.concat(Array.isArray(cur) ? flat(cur) : cur);
  }, []);
};
// [1, 2, 3, 4, 1, 2, 3, 1, 2, 3, 1, 2, 3, 5];

var arr = [1,2,3,4,4,1,1,2,1,1,1];
//通过ES6 Set函数的属性
function uniqe(arr){
  return [...new Set(arr)];
}
//用indexOf查找筛选
function uniqe1(arr){
  let newArr = [];
  for(let i in arr){
    if(newArr.indexOf(arr[i])==-1)
    {
      newArr.push(arr[i]);
    }
  }
  return newArr;
}
// 双指针，通过splice删除元素
function uniqe2(arr){
  let len = arr.length;
  for(let i=0;i<len;i++)
  {
    for(let j=i+1;j<len;j++)
    {
      if(arr[i]===arr[j])
      {
        arr.splice(j,1);
        j--;
        len--;
      }
    }
  }
  return arr;
}
// 先排序，用栈实现
function uniqe3(arr){
  let stack =[];
  arr.sort();
  let end = arr[0];
  stack.push(arr[0]);
  for(let i=1;i<arr.length;i++)
  {
    if(arr[i]!=end)
    {
      stack.push(arr[i]);
      end = arr[i];
    }
  }
  return stack;
}
//通过filter函数 a是数组里的元素，b是数组里每个元素第一次出现的index，c代表数组名。
let lms2 = arr.filter(function(a,b,c){
  return c.indexOf(a) === b;
})
let lms = uniqe3(arr);

事件机制/EventLoop（轮询）

javascirpt是单线程的，如果事件过多，过大就会导致阻塞。

所以提出了事件轮询机制，有主线程和事件副线程。主线程发送事件执行命令到副线程执行，执行完成向主线程返回结果。这样主线程就不会阻塞了。

原生的自定义事件

CreateEvent(eventType)；

作用：用于创建事件
eventType: HTMLEvents || MouseEvents || UIEvents (想获取的 Event 对象的事件模块名。)
返回值： 新创建的 Event 对象，具有指定的类型。

event.initEvent(eventType,canBubble,cancelable)

作用：用于初始化事件
eventType： 字符串值。事件的类型。
canBubble： 事件是否起泡
cancelable： 是否可以用 preventDefault() 方法取消事件

dispatchEvent(eventType)

作用：用于触发自定义事件
eventType： 要分派的 Event 对象。
